function createCachedObject(obj) {
  const cache = new Map();
  
  return new Proxy(obj, {
    get(target, property) {
      // –ï—Å–ª–∏ —ç—Ç–æ –º–µ—Ç–æ–¥ –∏ –æ–Ω –µ—â–µ –Ω–µ –∫–µ—à–∏—Ä–æ–≤–∞–Ω

          const key = JSON.stringify(args);
          
          if (methodCache.has(key)) {
            console.log(`üìã –í–æ–∑–≤—Ä–∞—â–∞–µ–º –∏–∑ –∫–µ—à–∞: ${property}(${args})`);

          return result;

        return cachedMethod;
      }
      
      return cache.get(property) || target[property];
    }
  });
}
